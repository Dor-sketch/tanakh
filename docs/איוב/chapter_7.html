איוב פרק ז'

א'הֲלֹֽא־צָבָ֣א לֶֽאֱנ֣וֹשׁ (על) [עֲלֵי־]אָ֑רֶץ וְכִימֵ֖י שָׂכִ֣יר יָמָֽיו׃

ב'כְּעֶ֥בֶד יִשְׁאַף־צֵ֑ל וּ֝כְשָׂכִ֗יר יְקַוֶּ֥ה פׇעֳלֽוֹ׃

ג'כֵּ֤ן הׇנְחַ֣לְתִּי לִ֭י יַרְחֵי־שָׁ֑וְא וְלֵיל֥וֹת עָ֝מָ֗ל מִנּוּ־לִֽי׃

ד'אִם־שָׁכַ֗בְתִּי וְאָמַ֗רְתִּי מָתַ֣י אָ֭קוּם וּמִדַּד־עָ֑רֶב וְשָׂבַ֖עְתִּי נְדֻדִ֣ים עֲדֵי־נָֽשֶׁף׃

ה'לָ֘בַ֤שׁ בְּשָׂרִ֣י רִ֭מָּה (וגיש) [וְג֣וּשׁ] עָפָ֑ר עוֹרִ֥י רָ֝גַ֗ע וַיִּמָּאֵֽס׃

ו'יָמַ֣י קַ֭לּוּ מִנִּי־אָ֑רֶג וַ֝יִּכְל֗וּ בְּאֶ֣פֶס תִּקְוָֽה׃

ז'זְ֭כֹר כִּי־ר֣וּחַ חַיָּ֑י לֹֽא־תָשׁ֥וּב עֵ֝ינִ֗י לִרְא֥וֹת טֽוֹב׃

ח'לֹא־תְ֭שׁוּרֵנִי עֵ֣ין רֹ֑אִי עֵינֶ֖יךָ בִּ֣י וְאֵינֶֽנִּי׃

ט'כָּלָ֣ה עָ֭נָן וַיֵּלַ֑ךְ כֵּ֥ן יוֹרֵ֥ד שְׁ֝א֗וֹל לֹ֣א יַעֲלֶֽה׃

י'לֹא־יָשׁ֣וּב ע֣וֹד לְבֵית֑וֹ וְלֹא־יַכִּירֶ֖נּוּ ע֣וֹד מְקֹמֽוֹ׃

י"אגַּם־אֲנִי֮ לֹ֤א אֶחֱשׇׂ֫ךְ־פִּ֥י אֲֽ֭דַבְּרָה בְּצַ֣ר רוּחִ֑י אָ֝שִׂ֗יחָה בְּמַ֣ר נַפְשִֽׁי׃

י"בהֲֽיָם־אָ֭נִי אִם־תַּנִּ֑ין כִּֽי־תָשִׂ֖ים עָלַ֣י מִשְׁמָֽר׃

י"גכִּֽי־אָ֭מַרְתִּי תְּנַחֲמֵ֣נִי עַרְשִׂ֑י יִשָּׂ֥א בְ֝שִׂיחִ֗י מִשְׁכָּבִֽי׃

י"דוְחִתַּתַּ֥נִי בַחֲלֹמ֑וֹת וּֽמֵחֶזְיֹנ֥וֹת תְּבַעֲתַֽנִּי׃

ט"ווַתִּבְחַ֣ר מַחֲנָ֣ק נַפְשִׁ֑י מָ֝֗וֶת מֵעַצְמוֹתָֽי׃

ט"זמָ֭אַסְתִּי לֹא־לְעֹלָ֣ם אֶחְיֶ֑ה חֲדַ֥ל מִ֝מֶּ֗נִּי כִּי־הֶ֥בֶל יָמָֽי׃

י"זמָֽה־אֱ֭נוֹשׁ כִּ֣י תְגַדְּלֶ֑נּוּ וְכִֽי־תָשִׁ֖ית אֵלָ֣יו לִבֶּֽךָ׃

י"חוַתִּפְקְדֶ֥נּוּ לִבְקָרִ֑ים לִ֝רְגָעִ֗ים תִּבְחָנֶֽנּוּ׃

י"טכַּ֭מָּה לֹא־תִשְׁעֶ֣ה מִמֶּ֑נִּי לֹֽא־תַ֝רְפֵּ֗נִי עַד־בִּלְעִ֥י רֻקִּֽי׃

כ'חָטָ֡אתִי מָ֤ה אֶפְעַ֨ל ׀ לָךְ֮ נֹצֵ֢ר הָ֫אָדָ֥ם לָ֤מָה שַׂמְתַּ֣נִי לְמִפְגָּ֣ע לָ֑ךְ וָאֶהְיֶ֖ה עָלַ֣י לְמַשָּֽׂא׃

כ"אוּמֶ֤ה ׀ לֹֽא־תִשָּׂ֣א פִשְׁעִי֮ וְתַעֲבִ֢יר אֶת־עֲוֺ֫נִ֥י כִּֽי־עַ֭תָּה לֶעָפָ֣ר אֶשְׁכָּ֑ב וְשִׁ֖חַרְתַּ֣נִי וְאֵינֶֽנִּי׃ {פ}
